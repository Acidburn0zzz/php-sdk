389:     if(in_array('curl', get_loaded_extensions())){
390:         $curl_handle = curl_init();
391:         curl_setopt($curl_handle, CURLOPT_URL, $ValidateUrl);
392:         curl_setopt($curl_handle, CURLOPT_CONNECTTIMEOUT, 5);
393:         curl_setopt($curl_handle, CURLOPT_TIMEOUT, 5);
394:         curl_setopt($curl_handle, CURLOPT_ENCODING, 'json');
395:         curl_setopt($curl_handle, CURLOPT_SSL_VERIFYPEER, false);
396:         if($post){
397:             curl_setopt($curl_handle, CURLOPT_POST, 1);
398:             curl_setopt($curl_handle, CURLOPT_POSTFIELDS, '');
399:         }
400:         if(ini_get('open_basedir') == '' && (ini_get('safe_mode') == 'Off' or !ini_get('safe_mode'))){
401:             curl_setopt($curl_handle, CURLOPT_FOLLOWLOCATION, 1);
402:             curl_setopt($curl_handle, CURLOPT_RETURNTRANSFER, true);
403:             $JsonResponse = curl_exec($curl_handle);
404:         }else{
405:             curl_setopt($curl_handle, CURLOPT_HEADER, 1);
406:             $url = curl_getinfo($curl_handle, CURLINFO_EFFECTIVE_URL);
407:             curl_close($curl_handle);
408:             $ch = curl_init();
409:             $url = str_replace('?','/?',$url);
410:             curl_setopt($ch, CURLOPT_URL, $url);
411:             curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
412:             $JsonResponse = curl_exec($ch);
413:             curl_close($ch);
414:         }
415:     }else{
416:         if($post){
417:             $postdata = http_build_query(
418:                 array(
419:                     'var1' => 'val'
420:                 )
421:             );
422:             $options = array('http' =>
423:                 array(
424:                     'method'  => 'POST',
425:                     'timeout' => 10,
426:                     'header'  => 'Content-type: application/x-www-form-urlencoded',
427:                     'content' => $postdata
428:                 )
429:             );
430:             $context  = stream_context_create($options);
431:         }else{
432:             $context = NULL;
433:         }
